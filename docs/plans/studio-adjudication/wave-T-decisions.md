  export type ApiErrorKind = 'network'|'server'|'rate_limit'|'auth'|'forbidden'|'not_found'|'conflict'|'invalid'|'unknown';
  export type ClassifiedApiError = { kind; retryable: boolean; status?: number; messageKey: string; detail?: string };
  export function classifyApiError(err: unknown): ClassifiedApiError

RULES (all derivable from api.ts's throw contract):
- `status = (err as {status?:number}).status`, `code = (err as {code?:string}).code`.
- retryable = (status === undefined) || status >= 500 || status === 429.
- kind: undefined status -> 'network'; 429 -> 'rate_limit'; >=500 -> 'server'; 401 -> 'auth'; 403 -> 'forbidden'; 404 -> 'not_found'; 409/412 -> 'conflict'; 400/422 -> 'invalid'; else 'unknown'.
- **detail (THE T4 GUARD): return `undefined` when `code === 'PARSE_ERROR'` OR `status === undefined`.** Otherwise `readBackendError(err)`. Rationale grounded in code: a non-JSON body means the response never came from a backend handler (dev-proxy/nginx/gateway transport failure) — api.ts:102 stuffs the raw text into `{code:'PARSE_ERROR', message: text}`, which is exactly how the proxy string reached the user in T4. Gate on `code` BEFORE calling readBackendError, since readBackendError falls back to `err.message`.
- messageKey -> `errors.<kind>` in `frontend/src/i18n/locales/en/common.json` (shared by studio + translation): errors.network "Can't reach the server.", errors.server "The translation service is unavailable.", errors.rate_limit "Too many requests — retrying shortly.", errors.forbidden "You don't have access to this book's translations.", errors.not_found, errors.conflict, errors.invalid, errors.unknown.
